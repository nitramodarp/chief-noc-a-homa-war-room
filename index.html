// ============================================================ // HKB_Scraper.gs — HarryKnowsBall Dynasty Value Scraper // Populates column Y (HKB_VALUE) in your PROJECTIONS sheet // // SETUP: // 1. Open your Google Sheet → Extensions → Apps Script // 2. Paste this file, Save, Run scrapeHKBValues // 3. Authorize when prompted // 4. Optionally run createWeeklyTrigger for auto-updates // ============================================================ const SHEET_NAME = 'PROJECTIONS'; const HKB_COL = 25; // Column Y const NAME_COL = 2; // Column B = player name const HEADER_ROW = 1; // Position and level tabs — each returns top ~50 for that filter // Union of all tabs covers the full player pool const HKB_URLS = [ 'https://harryknowsball.com/rankings', 'https://harryknowsball.com/rankings?positions=C', 'https://harryknowsball.com/rankings?positions=1B', 'https://harryknowsball.com/rankings?positions=2B', 'https://harryknowsball.com/rankings?positions=3B', 'https://harryknowsball.com/rankings?positions=SS', 'https://harryknowsball.com/rankings?positions=OF', 'https://harryknowsball.com/rankings?positions=SP', 'https://harryknowsball.com/rankings?positions=RP', 'https://harryknowsball.com/rankings?level=Prospects', 'https://harryknowsball.com/rankings?level=MLB', 'https://harryknowsball.com/rankings?level=MiLB', ]; // ── Name normalization ───────────────────────────────────── function cleanName(name) { if (!name) return ''; return name.toString() .normalize('NFD') .replace(/[\u0300-\u036f]/g, '') // strip accents: é→e .toLowerCase() .replace(/[''`´\.]/g, '') // remove apostrophes/periods .replace(/[^a-z0-9 ]/g, ' ') // everything else → space .replace(/\s+/g, ' ') .trim(); } function nameVariants(name) { const base = cleanName(name); const stripped = base .replace(/\b(jr|sr|ii|iii|iv)\b/g, '') .replace(/\s+/g, ' ').trim(); return stripped === base ? [base] : [base, stripped]; } // ── Core parser — PROVEN WORKING from testFetch ──────────── // Key insight: strip ALL html tags first, then split on ‧ (U+2027) // The plaintext format is: "...Player Name‧LEVEL‧TEAM · age · pos · VALUE..." function parseHKBHtml(html) { const values = {}; const HYPHEN = '\u2027'; // ‧ // Step 1: Strip HTML entirely to get clean plaintext const text = html .replace(/]*>[\s\S]*?<\/script>/gi, '') .replace(/]*>[\s\S]*?<\/style>/gi, '') .replace(/<[^>]+>/g, ' ') .replace(/&/g, '&').replace(/</g, '<').replace(/>/g, '>') .replace(/ /g, ' ').replace(/'/g, "'") .replace(/\s+/g, ' '); if (!text.includes(HYPHEN)) return values; // Step 2: Split on ‧ — each player entry spans two segments: // segments[i] ends with "...Player Name" // segments[i+1] starts with "LEVEL‧TEAM · age · pos · VALUE · ..." const segments = text.split(HYPHEN); for (let i = 0; i < segments.length - 1; i++) { // ── Extract name from end of segments[i] ────────────── const namePart = segments[i].trim(); // Take up to last 4 words — the player name const rawWords = namePart.split(' ').filter(w => w.length > 0); const nameWords = rawWords .slice(-4) .join(' ') .replace(/[^a-zA-ZÀ-ÿ\s'.\-]/g, '') .trim(); if (!nameWords || nameWords.length < 3) continue; // ── Extract value from segments[i+1] and optionally [i+2] ─ const valuePart = (segments[i+1] + ' ' + (segments[i+2] || '')).substring(0, 150); // Dynasty values are comma-formatted integers 100-10000 // e.g. "10,000" or "1,234" — NOT decimals like "25.7" (age) const valueMatch = valuePart.match(/\b(\d{1,2},\d{3})\b/); if (!valueMatch) continue; const dynasty = parseInt(valueMatch[1].replace(',', '')); if (dynasty < 100 || dynasty > 10000) continue; // ── Store both full name and suffix-stripped variant ─── const cn = cleanName(nameWords); if (cn.length > 2 && !values[cn]) { values[cn] = dynasty; } const stripped = cn.replace(/\b(jr|sr|ii|iii|iv)\b/g, '').replace(/\s+/g,' ').trim(); if (stripped !== cn && stripped.length > 2 && !values[stripped]) { values[stripped] = dynasty; } } return values; } // ── Fetch one URL ───────────────────────────────────────── function fetchOne(url) { try { const resp = UrlFetchApp.fetch(url, { muteHttpExceptions: true, headers: { 'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36', 'Accept': 'text/html,application/xhtml+xml', }, followRedirects: true, }); if (resp.getResponseCode() !== 200) { Logger.log(` HTTP ${resp.getResponseCode()} — ${url}`); return {}; } return parseHKBHtml(resp.getContentText()); } catch(e) { Logger.log(` Error on ${url}: ${e.message}`); return {}; } } // ── Main: scrape all tabs and write to sheet ─────────────── function scrapeHKBValues() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { SpreadsheetApp.getUi().alert(`Sheet "${SHEET_NAME}" not found.`); return; } Logger.log('Scraping HKB across all position tabs...'); const all = {}; for (let i = 0; i < HKB_URLS.length; i++) { const url = HKB_URLS[i]; const vals = fetchOne(url); let added = 0; for (const [k,v] of Object.entries(vals)) { if (!all[k]) { all[k] = v; added++; } } const label = url.includes('?') ? url.split('?')[1] : 'all'; Logger.log(` [${label}] ${Object.keys(vals).length} parsed, ${added} new → total ${Object.keys(all).length}`); if (i < HKB_URLS.length - 1) Utilities.sleep(1000); } Logger.log(`\nTotal unique players scraped: ${Object.keys(all).length}`); if (Object.keys(all).length < 20) { SpreadsheetApp.getUi().alert( '⚠️ Only ' + Object.keys(all).length + ' players found.\n\n' + 'HKB page structure may have changed.\nRun testFetch() and check logs to debug.' ); return; } writeToSheet(sheet, all); } // ── Write values to column Y ────────────────────────────── function writeToSheet(sheet, hkbValues) { const lastRow = sheet.getLastRow(); if (lastRow <= HEADER_ROW) return; sheet.getRange(HEADER_ROW, HKB_COL).setValue('HKB_VALUE'); const names = sheet .getRange(HEADER_ROW + 1, NAME_COL, lastRow - HEADER_ROW, 1) .getValues(); const updates = []; let matched = 0; const unmatched = []; for (let i = 0; i < names.length; i++) { const raw = (names[i][0] || '').toString().trim(); if (!raw) { updates.push(['']); continue; } let value = 0; // Try exact + suffix-stripped variants for (const variant of nameVariants(raw)) { if (hkbValues[variant]) { value = hkbValues[variant]; break; } } // Fuzzy prefix match as fallback if (!value) { const cn = cleanName(raw); for (const [key, val] of Object.entries(hkbValues)) { if (key.length > 6 && (cn.startsWith(key) || key.startsWith(cn))) { value = val; break; } } } if (value > 0) { matched++; updates.push([value]); } else { unmatched.push(raw); updates.push(['']); } } // Single batch write sheet.getRange(HEADER_ROW + 1, HKB_COL, updates.length, 1).setValues(updates); SpreadsheetApp.getActiveSpreadsheet().toast(`HKB updated: ${matched} players matched`, '✅ Done', 5); const msg = `✅ HKB Update Complete!\n\n` + `Scraped from HKB: ${Object.keys(hkbValues).length} players\n` + `Matched to your sheet: ${matched}\n` + `Unmatched: ${unmatched.length}\n\n` + (unmatched.length > 0 ? `First 15 unmatched:\n${unmatched.slice(0,15).join('\n')}` : '🎉 All players matched!'); Logger.log(msg); SpreadsheetApp.getUi().alert(msg); } // ── Quick test — no writes ──────────────────────────────── function testFetch() { Logger.log('=== TEST: main page only ==='); const vals = fetchOne('https://harryknowsball.com/rankings'); Logger.log(`Found ${Object.keys(vals).length} players`); Object.entries(vals).slice(0, 20).forEach(([k,v]) => Logger.log(` ${k}: ${v}`)); } // ── Full test across all tabs — no writes ───────────────── function testAllTabs() { Logger.log('=== TEST ALL TABS ==='); const all = {}; for (const url of HKB_URLS) { const vals = fetchOne(url); let n = 0; for (const [k,v] of Object.entries(vals)) { if (!all[k]) { all[k] = v; n++; } } const label = url.includes('?') ? url.split('?')[1] : 'all'; Logger.log(` [${label}] ${Object.keys(vals).length} parsed, ${n} new → total ${Object.keys(all).length}`); Utilities.sleep(800); } Logger.log(`\nGrand total: ${Object.keys(all).length} unique players`); Logger.log('Sample (first 30):'); Object.entries(all).slice(0,30).forEach(([k,v]) => Logger.log(` ${k}: ${v}`)); } // ── Weekly auto-trigger ─────────────────────────────────── function createWeeklyTrigger() { ScriptApp.getProjectTriggers() .filter(t => t.getHandlerFunction() === 'scrapeHKBValues') .forEach(t => ScriptApp.deleteTrigger(t)); ScriptApp.newTrigger('scrapeHKBValues') .timeBased().onWeekDay(ScriptApp.WeekDay.MONDAY).atHour(9).create(); SpreadsheetApp.getUi().alert('✅ Auto-trigger set — HKB updates every Monday 9am.'); }
